## 조합 ##
```cpp
#include <iostream>
using namespace std;

int arr[1001][1001];

int main(){
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    
    int N,K; cin>>N>>K;
    
    for(int i=0; i<=N; i++){
        for(int k=0; k<=K; k++){
            if(i==k||k==0)arr[i][k]=1;
            else arr[i][k]=(arr[i-1][k-1]+arr[i-1][k]);
        }
    }
    cout<<arr[N][K];
}
```
